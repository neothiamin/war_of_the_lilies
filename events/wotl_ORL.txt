#Conseil Rule Breaking Down
country_event = {
    id = wotl_orl.2
    title = wotl_orl.2.t
    desc = wotl_orl.2.desc
    picture = OVEREXTENSION_eventPicture

    fire_only_once = yes

    trigger = {
        tag = ORL
        is_subject = no
        has_reform = orl_council_reform
        is_year = 1438
    }

    immediate = {
        set_country_flag = orl_conseil_breakdown
    }

    mean_time_to_happen = {
        months = 120
        modifier = {
            factor = 0.8
            is_at_war = yes
        }
    }

    #Arrange a grand parade celebrating the Conseil.
    option = {
        name = wotl_orl.2.a
        add_years_of_income = -0.5
        random_list = {
            25 = {
                custom_tooltip = wotl_orl.2.a.1.tt
                hidden_effect = {
                    country_event = {
                        id = wotl_orl.3
                        days = 30
                    }
                }
            }
            75 = {
                custom_tooltip = wotl_orl.2.a.2.tt
                hidden_effect = {
                    country_event = {
                        id = wotl_orl.4
                        days = 30
                    }
                }
            }
        }
        ai_chance = {
            factor = 25
        }
    }

    #Reassure the populace that the system is stable.
    option = {
        name = wotl_orl.2.b
        add_prestige = -10
        capital_scope = {
            add_unrest = 4
        }
        set_country_flag = wotl_orl_reassure
        ai_chance = {
            factor = 40
        }
    }

    #What hogwash! Don't they know that the Conseil rules, not the peasantry?
    option = {
        name = wotl_orl.2.c
        capital_scope = {
            add_unrest = 8
        }
        ai_chance = {
            factor = 25
        }
        set_country_flag = wotl_orl_ignored_peasants
    }

    #Consolidate central power.
    option = {
        name = wotl_orl.2.d
        add_adm_power = -100
        add_country_modifier = {
            name = wotl_orl_consolidated_power
            duration = -1
        }
        ai_chance = {
            factor = 10
        }
    }
}

#Monarchist Rebellion
country_event = {
    id = wotl_orl.5
    title = wotl_orl.5.t
    desc = wotl_orl.5
    picture = ANGRY_MOB_eventPicture

    fire_only_once = yes

    trigger = {
        tag = ORL
        has_reform = orl_council_reform
        is_subject = no
        had_country_flag = {
            flag = orl_conseil_breakdown
            days = 3650
        }
        NOT = {
            has_global_flag = wotl_war_over
        }
    }

    mean_time_to_happen = {
        months = 90
    }

    #The republic must live on!
    option = {
        name = wotl_orl.5.a
        capital_scope = {
            spawn_rebels = {
                size = 2
                type = monarchist
            }
        }
        add_stability = -1
    }

    #Crown [Root.Monarch.GetName] as King.
    option = {
        name = wotl_orl.5.b
        change_government = monarchy
        add_stability = -1
        set_government_rank = 2
    }
}
