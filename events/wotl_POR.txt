#Henrique, o Navegador

country_event = {
    id = wotl_por.1
    title = wotl_por.1.t
    desc = wotl_por.1.desc
    picture = EXPLORERS_eventPicture

    fire_only_once = yes

    mean_time_to_happen = {
        months = 12
    }

    trigger = {
        tag = POR
        current_age = age_of_discovery
        any_owned_province = {
            has_port = yes
        }
        NOT = {
            has_idea_group = exploration_ideas
        }
        is_at_war = no
        owns_core_province = 230
        dip_tech = 3
    }

    #Give Henrique whatever he needs!
    option = {
        name = wotl_por.1.a
        define_explorer = {
            shock = 2
            fire = 1
            manuever = 4
            name = "Henrique, o Navegador"
        }
        add_country_modifier = {
            name = wotl_por_henrique
            duration = 9125
        }
    }

    #Grant Henrique whatever we can spare.
    option = {
        name = wotl_por.1.b
        set_country_flag = wotl_por_henrique_must_prove_self
        define_explorer = {
            shock = 1
            fire = 0
            manuever = 2
            name = "Henrique, o Navegador"
        }
    }
    
}
