namespace = wotl_proper

#War of the Lilies!
country_event = {
    id = wotl_proper.1
    title = wotl_proper.1.t
    desc = wotl_proper.1.desc
    picture = ANGRY_MOB_eventPicture

    is_triggered_only = yes

    fire_only_once = yes

    trigger = {
        tag = ENG
        religion_group = christian
        NOT = {
            has_global_flag = wotl_war_over
        }
        normal_or_historical_nations = yes
    }

    #Dig in! We will win this!
    option = {
        name = wotl_proper.1.a
        set_global_flag = wotl_war_active
        add_stability = -1
        if = {
            limit = {
                FRA = {
                    NOT = {
                        war_with = ENG
                    }
                }
            }
            FRA = {
                country_event = {
                    id = wotl_proper.2
                }
            }
        }
    }
}

#War of the Lilies! (French)
country_event = {
    id = wotl_proper.2
    title = wotl_proper.2.t
    desc = wotl_proper.2.desc
    picture = GOOD_WITH_MONARCH_eventPicture

    is_triggered_only = yes

    fire_only_once = yes

    #Aux armes! Down with the English! France for the French!
    option = {
        name = wotl_proper.2.a
        declare_war_with_cb = {
            casus_belli = cb_independence_war
            who = ENG
        }
        add_country_modifier = {
            name = wotl_fra_independence
            duration = 3650
        }
        if = {
            limit = {
                exists = ORL
                ORL = {
                    has_country_flag = wotl_orl_found_heir
                }
            }
            ORL = {
                country_event = {
                    id = wotl_proper.3
                    days = 15
                    random = 5
                }
            }
        }
    }

    #We must not provoke our overlord.
    option = {
        name = wotl_proper.2.b
        trigger = {
            ai = no
        }
        add_prestige = -5
        custom_tooltip = wotl_proper.2.b.tt
        ENG = {
            set_country_flag = wotl_eng_won_war
        }
    }
}

    title = wotl_proper.4.t
    desc = wotl_proper.4.desc
    picture = GOOD_WITH_MONARCH_eventPicture

    is_triggered_only = yes

    fire_only_once = yes

    #Accept.
    option = {
        name = wotl_proper.4.a
        ai_chance = {
            factor = 75
        }
        hidden_effect = {
            ORL = {
                exile_ruler_as = wotl_orl_king_of_france
            }
        }
        change_adm = 1
        change_dip = 1
        change_mil = 1
        set_ruler = wotl_orl_king_of_france
        create_union = ORL
        if = {
            limit = {
                ORL = {
                    ai = no
                }
            }
            ORL = {
                country_event = {
                    id = wotl_proper.5
                    days = 15
                    random = 5
                }
            }
        }
    }

    #Decline.
    option = {
        name = wotl_proper.4.b
        ORL = {
            add_prestige = -10
        }
    }

}

#FRA - Walloon Support!
country_event = {
    id = wotl_proper.7
    title = wotl_proper.7.t
    desc = wotl_proper.7.desc
    picture = GOOD_WITH_MONARCH_eventPicture

    trigger = {
        tag = FRA
        has_global_flag = wotl_war_active
        any_owned_province = {
            culture = wallonian
            controlled_by = ROOT
        }
    }

    mean_time_to_happen = {
        months = 18
    }

    immediate = {
        set_saved_name = {
            key = fra_walloon_general
            type = advisor
            scope = ROOT
        }
    }

    #We welcome them as brothers in arms!
    option = {
        name = wotl_proper.7.a
        set_country_flag = wotl_fra_walloon_support_yes
        custom_tooltip = wotl_proper.7.a.tt
        random_owned_province = {
            limit = {
                culture = wallonian
                controlled_by = ROOT
            }
            infantry = THIS
            infantry = THIS
            infantry = THIS
            infantry = THIS
            infantry = THIS
            cavalry = THIS
            cavalry = THIS
            cavalry = THIS
        }
        define_general = {
            shock = 3
            fire = 1
            manuever = 2
            siege = 1
            trait = master_of_arms_personality
            name = fra_walloon_general
        }
    }

    #We would love to, but we don't need their help.
    option = {
        name = wotl_proper.7.b
        every_owned_province = {
            limit = {
                culture = wallonian
            }
            add_unrest = 4
        }
    }
}

#If France declares an independence war before the WOTL happens, this event makes the disaster fire for England
country_event = {
    id = wotl_proper.12
    title = wotl_proper.2.t
    desc = wotl_proper.2.desc
    picture = ANGRY_MOB_eventPicture

    fire_only_once = yes

    trigger = {
        tag = FRA
        war_with = ENG
        NOT = {
            has_global_flag = wotl_war_active
            has_global_flag = wotl_war_over
        }
    }

    mean_time_to_happen = {
        days = 5
    }

    #Aux armes! Down with the English! France for the French!
    option = {
        name = wotl_proper.2.a
        ENG = {
            add_disaster_progress = {
                value = 100
                disaster = war_of_the_lilies_disaster
            }
        }
        add_country_modifier = {
            name = wotl_fra_independence
            duration = 3650
        }
        if = {
            limit = {
                exists = ORL
                ORL = {
                    has_country_flag = wotl_orl_found_heir
                }
            }
            ORL = {
                country_event = {
                    id = wotl_proper.3
                    days = 15
                    random = 5
                }
            }
        }
    }
}
